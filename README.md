#include <iostream>
using namespace std;

int main() {
    int a = 8, b = 3, c = 2;  // hard-coded values

    cout << "a + b * c = " << (a + b * c) << endl;
    cout << "(a + b) * c = " << ((a + b) * c) << endl;
    cout << "a / b * c = " << (a / b * c) << endl;
    cout << "a / (b * c) = " << (a / (b * c)) << endl;

    return 0;
}
